import cv2
import mediapipe as mp
import RPi.GPIO as GPIO
import time

# ---------------- GPIO SETUP -----------------
GPIO.setmode(GPIO.BCM)

# Index, Middle, Ring, Pinky
relay_pins = [17, 27, 22, 23]

for pin in relay_pins:
    GPIO.setup(pin, GPIO.OUT)
    GPIO.output(pin, GPIO.HIGH)   # OFF (Active LOW)

# ---------------- MEDIAPIPE SETUP -----------------
mp_hands = mp.solutions.hands
mp_draw = mp.solutions.drawing_utils

hands = mp_hands.Hands(
    max_num_hands=1,
    min_detection_confidence=0.7,
    min_tracking_confidence=0.7
)

# ---------------- VIDEO CAPTURE -----------------
cap = cv2.VideoCapture(0)
cap.set(3, 640)
cap.set(4, 480)

# ---------------- FINGER DETECTION -----------------
def get_finger_states(lm):
    fingers = [0, 0, 0, 0]  # Index, Middle, Ring, Pinky

    # Index
    if lm[8].y < lm[6].y:
        fingers[0] = 1

    # Middle
    if lm[12].y < lm[10].y:
        fingers[1] = 1

    # Ring
    if lm[16].y < lm[14].y:
        fingers[2] = 1

    # Pinky
    if lm[20].y < lm[18].y:
        fingers[3] = 1

    return fingers

# ---------------- MAIN LOOP -----------------
prev_states = [0, 0, 0, 0]

while True:
    ret, frame = cap.read()

    if not ret:
        print("Camera Not Found!")
        break

    frame = cv2.flip(frame, 1)
    rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    results = hands.process(rgb)

    if results.multi_hand_landmarks:
        for handLms in results.multi_hand_landmarks:
            mp_draw.draw_landmarks(
                frame,
                handLms,
                mp_hands.HAND_CONNECTIONS
            )

            lm = handLms.landmark
            finger_states = get_finger_states(lm)

            # ----------- STABLE CONTROL ------------
            if finger_states != prev_states:
                prev_states = finger_states

                for i in range(4):
                    if finger_states[i] == 1:
                        GPIO.output(relay_pins[i], GPIO.LOW)   # ON
                    else:
                        GPIO.output(relay_pins[i], GPIO.HIGH)  # OFF

            # Display
            cv2.putText(
                frame,
                f"I:{finger_states[0]} M:{finger_states[1]} "
                f"R:{finger_states[2]} P:{finger_states[3]}",
                (10, 50),
                cv2.FONT_HERSHEY_SIMPLEX,
                1,
                (0, 255, 0),
                3
            )

    cv2.imshow("4 Finger Home Automation", frame)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

    time.sleep(0.05)

# ---------------- CLEANUP -----------------
cap.release()
cv2.destroyAllWindows()
GPIO.cleanup()
